package main

import (
	"fmt"
	"strconv"
	"strings"
)

func isDiffValidForCode(luauCode string, unifiedDiff string) bool {
	// Split the code and diff into lines
	codeLines := strings.Split(luauCode, "\n")
	diffLines := strings.Split(unifiedDiff, "\n")

	// Basic validation of diff format
	if len(diffLines) < 3 { // Minimum: header + hunk header + one change
		return false
	}

	// Process each line of the diff
	var currentLine int
	for i, line := range diffLines {
		// Check diff header (first two lines)
		if i == 0 {
			if !strings.HasPrefix(line, "---") {
				fmt.Println("first line doesn't start with ---")
				return false
			}
			continue
		}
		if i == 1 {
			if !strings.HasPrefix(line, "+++") {
				fmt.Println("second line doesn't start with +++")
				return false
			}
			continue
		}

		// Process hunk headers (e.g., @@ -1,3 +1,4 @@)
		if strings.HasPrefix(line, "@@") {
			parts := strings.Split(line, " ")
			if len(parts) < 3 {
				return false
			}

			// Parse original file range: -start,count
			origRange := strings.TrimPrefix(parts[1], "-")
			origParts := strings.Split(origRange, ",")
			if len(origParts) != 2 {
				return false
			}

			// Convert start line to 0-based index
			startLine, err := strconv.Atoi(origParts[0])
			if err != nil {
				return false
			}
			startLine-- // Convert to 0-based index

			count, err := strconv.Atoi(origParts[1])
			if err != nil {
				return false
			}

			// Check if the range is valid for the code
			if startLine < 0 || startLine >= len(codeLines) || startLine+count > len(codeLines) {
				return false
			}

			currentLine = startLine
			continue
		}

		// Validate context and removal lines match the original code
		if strings.HasPrefix(line, " ") || strings.HasPrefix(line, "-") {
			if currentLine >= len(codeLines) {
				return false
			}
			expected := strings.TrimPrefix(line, " ")
			expected = strings.TrimPrefix(expected, "-")
			if codeLines[currentLine] != expected {
				return false
			}
			currentLine++
		}
	}

	return true
}
